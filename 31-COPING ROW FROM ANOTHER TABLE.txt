COPYING ROW FROM ANOTHER TABLE


INSERT INTO emp(EMP_ID,EMAIL,SALARY,DEPARTMENT_ID,NAME,DEPARTMENT_NAME)

SELECT EMPLOYEE_ID,EMAIL,SALARY,DEPARTMENT_ID,LAST_NAME,DEPARTMENT_NAME
FROM EMPLOYEES NATURAL JOIN DEPARTMENTS
WHERE DEPARTMENT_ID=90;

select*
from emp
ORDER BY EMP_ID;

ROLLBACK INTO emp(EMP_ID,EMAIL,SALARY,DEPARTMENT_ID,NAME,DEPARTMENT_NAME)
